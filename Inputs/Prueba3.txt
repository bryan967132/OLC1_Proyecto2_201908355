int ack(int m, int n){
    if (m == 0) {
        return n + 1;
    } else if (n == 0) {
        return ack(m - 1, 1);
    } else {
        return ack(m - 1, ack(m, n - 1));
    }
}


void inicio(){
  
    print("Ackerman de 3,0 "+ack(3,0));
    print("Ackerman de 2,8 "+ack(2,8));
    print("Ackerman de 2,1 "+ack(2,1));
}
